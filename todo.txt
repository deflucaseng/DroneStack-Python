☐ 4.2. Implement geometric controller (drone_stack/control/geometric_controller.py)
   
   This is complex! Key steps:
   1. Compute desired thrust vector from position error
   2. Compute desired attitude from thrust vector
   3. Compute attitude error
   4. Compute angular rate commands
   
   ```python
   # drone_stack/control/geometric_controller.py
   import numpy as np
   from drone_stack.core.pose import ControlOutput
   
   class GeometricController:
       def __init__(self, mass, Kp, Kd, Kp_att):
           self.mass = mass
           self.Kp = Kp  # Position gain
           self.Kd = Kd  # Velocity gain
           self.Kp_att = Kp_att  # Attitude gain
           self.g = 9.81
       
       def compute(self, desired_state, estimated_pose):
           """Compute control from desired trajectory state and current pose."""
           
           # 1. Position control -> desired thrust
           pos_error = desired_state.position - estimated_pose.position
           vel_error = desired_state.velocity - estimated_pose.velocity
           
           desired_accel = (
               self.Kp * pos_error + 
               self.Kd * vel_error + 
               desired_state.acceleration + 
               np.array([0, 0, self.g])
           )
           
           # 2. Thrust magnitude and direction
           thrust = self.mass * np.linalg.norm(desired_accel)
           z_body_desired = desired_accel / np.linalg.norm(desired_accel)
           
           # 3. Compute desired rotation matrix
           # (This is the tricky part - see Lee et al. paper)
           R_desired = self._compute_desired_rotation(z_body_desired, desired_state.yaw)
           
           # 4. Attitude error
           R_current = estimated_pose.orientation
           R_error = 0.5 * vee(R_desired.T @ R_current - R_current.T @ R_desired)
           
           # 5. Angular rate commands
           angular_rates = self.Kp_att * R_error
           
           # 6. Convert to roll/pitch/yaw rates and throttle
           throttle = thrust / (self.mass * self.g)
           
           return ControlOutput(
               roll_rate=angular_rates[0],
               pitch_rate=angular_rates[1],
               yaw_rate=angular_rates[2],
               throttle=throttle
           )
   ```

☐ 5.2. Implement polynomial representation (drone_stack/planning/polynomial.py)
   
   ```python
   # drone_stack/planning/polynomial.py
   from numba import jit
   import numpy as np
   
   @jit(nopython=True)
   def evaluate_polynomial(coeffs, t, order=7):
       """Evaluate position, velocity, acceleration from polynomial.
       
       coeffs: (order+1,) coefficients for x, y, or z
       t: time in segment
       """
       # Position: p(t) = c0 + c1*t + c2*t^2 + ... + c7*t^7
       pos = 0.0
       vel = 0.0
       acc = 0.0
       
       for i in range(order + 1):
           pos += coeffs[i] * (t ** i)
           if i > 0:
               vel += i * coeffs[i] * (t ** (i-1))
           if i > 1:
               acc += i * (i-1) * coeffs[i] * (t ** (i-2))
       
       return pos, vel, acc
   ```

☐ 5.3. Implement QP solver (drone_stack/planning/min_snap/qp_solver.py)
   
   This is the hardest part! You're solving:
   
   minimize: ∫ ||d^4 x/dt^4||^2 dt  (minimize snap)
   subject to: 
   - waypoint constraints (position)
   - continuity constraints (velocity, acceleration, jerk)
   - initial/final conditions
   
   ```python
   # drone_stack/planning/min_snap/qp_solver.py
   import numpy as np
   from scipy.linalg import solve
   
   class MinSnapQPSolver:
       def __init__(self, order=7):
           self.order = order
       
       def solve(self, waypoints, segment_times):
           """Solve min-snap QP for given waypoints.
           
           Returns: coefficients for each segment
           """
           n_segments = len(waypoints) - 1
           n_coeffs = (self.order + 1) * n_segments
           
           # Build cost matrix Q (minimize snap)
           Q = self._build_cost_matrix(segment_times)
           
           # Build constraint matrices A, b
           A, b = self._build_constraints(waypoints, segment_times)
           
           # Solve QP: min x^T Q x subject to Ax = b
           # This simplifies to solving: (A^T A)^-1 A^T b with Q weighting
           coeffs = solve(A.T @ A + 0.001 * Q, A.T @ b)
   ```

☐ 5.4. Implement trajectory class (drone_stack/planning/trajectory.py)
   
   ```python
   # drone_stack/planning/trajectory.py
   from drone_stack.core.pose import TrajectoryState
   from drone_stack.planning.polynomial import evaluate_polynomial
   
   class Trajectory:
       def __init__(self, coeffs_x, coeffs_y, coeffs_z, segment_times):
           """Store polynomial coefficients for each axis."""
           self.coeffs_x = coeffs_x
           self.coeffs_y = coeffs_y
           self.coeffs_z = coeffs_z
           self.segment_times = segment_times
           self.duration = sum(segment_times)
       
       def sample(self, t):
           """Sample trajectory at time t."""
           # Find which segment we're in
           segment_idx, t_in_segment = self._find_segment(t)
           
           # Evaluate polynomials
           x, vx, ax = evaluate_polynomial(self.coeffs_x[segment_idx], t_in_segment)
           y, vy, ay = evaluate_polynomial(self.coeffs_y[segment_idx], t_in_segment)
           z, vz, az = evaluate_polynomial(self.coeffs_z[segment_idx], t_in_segment)
           
           return TrajectoryState(
               position=np.array([x, y, z]),
               velocity=np.array([vx, vy, vz]),
               acceleration=np.array([ax, ay, az]),
               yaw=0.0  # TODO: Compute from velocity direction
           )
       
       def _find_segment(self, t):
           """Find which segment and local time."""
           cumsum = 0.0
           for i, dt in enumerate(self.segment_times):
               if t < cumsum + dt:
                   return i, t - cumsum
               cumsum += dt
           return len(self.segment_times) - 1, self.segment_times[-1]
   ```

☐ 5.5. Implement trajectory planner (drone_stack/planning/trajectory_planner.py)
   
   ```python
   # drone_stack/planning/trajectory_planner.py
   from drone_stack.planning.min_snap.qp_solver import MinSnapQPSolver
   from drone_stack.planning.trajectory import Trajectory
   
   class TrajectoryPlanner:
       def __init__(self, config):
           self.avg_speed = config['avg_speed']
           self.qp_solver = MinSnapQPSolver(order=7)
       
       def plan(self, start, gates):
           """Plan trajectory from start through gates."""
           # Build waypoints
           waypoints = [start]
           waypoints.extend([gate.position for gate in gates])
           
           # Allocate time for each segment
           segment_times = self._allocate_time(waypoints)
           
           # Solve min-snap for each axis
           coeffs_x = self.qp_solver.solve([w[0] for w in waypoints], segment_times)
           coeffs_y = self.qp_solver.solve([w[1] for w in waypoints], segment_times)
           coeffs_z = self.qp_solver.solve([w[2] for w in waypoints], segment_times)
           
           return Trajectory(coeffs_x, coeffs_y, coeffs_z, segment_times)
       
       def _allocate_time(self, waypoints):
           """Allocate time for each segment based on distance."""
           times = []
           for i in range(len(waypoints) - 1):
               dist = np.linalg.norm(waypoints[i+1] - waypoints[i])
               times.append(dist / self.avg_speed)
           return times
   ```

CRITICAL SUCCESS FACTORS
═══════════════════════════════════════════════════════════════════

1. TEST EARLY, TEST OFTEN
   - After each phase, verify it works before moving on
   - Don't accumulate bugs

2. START SIMPLE
   - PID before geometric control
   - Ground truth state before estimation
   - Straight lines before aggressive racing

3. LOG EVERYTHING
   - You can't fix what you can't measure
   - Plot data after every test

4. VERSION CONTROL
   - Commit after each working feature
   - Tag stable versions

5. SAFETY FIRST
   - Always test in simulation first
   - Start real hardware tests with conservative gains
   - Have emergency stop ready
